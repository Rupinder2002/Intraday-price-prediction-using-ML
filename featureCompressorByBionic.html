import pandas as pd
import numpy as np

data= pd.read_csv("iris.data", header=0, names=["sepal_width","sepal_length","petal_width","petal_length","class"])
c= data.drop("class",axis=1)

def FeatureCompressor(normalised_Feature_Matrix, FinalFeatures=2):
       
        '''
        This function provides you with the Compressed features using PCA technique 
        though exploiting the use of covariance matrix of the feature matrix.
        Important: Features matrics must be mean centralised and features should be normalized between [-1,1]

        Arguments: 
        normalised_Feature_Matrix (type: Numpy.Array Object)  ===> the features matrix of the dataset it can be of any Instancesxfeatures dimensions
        FinalFeatures (type: Int Object)              ===> the number of features vector to which data is planned to be reduced

        Return:
        returns (Information_preserved(type: float object) , transformation_mat (type: Numpy.Array Object))

        '''

        
        arr= np.array(normalised_Feature_Matrix)
        arr2= arr.transpose()
        covariance= np.dot(arr2,arr)
        e,v= np.linalg.eig(covariance)

        df= pd.DataFrame([e,v]).transpose()
        df.columns=(["EigenValue","EigenVector"])
        dfs=df.set_index('EigenValue')
        dfsorted=dfs.sort_index(ascending=False)

        v= np.array(dfsorted["EigenVector"])
        vs= np.zeros((len(v),len(v)))
        cs= np.zeros((len(v),len(v)))

        it=0;
        for i in v:
                vs[:,it]=i
                cs[it]=e
                it=it+1     
        csdiag=(np.diag(np.diag(cs)))

        vstrans= vs.transpose()
        Dtemp= np.dot(vstrans,csdiag)
        D= np.dot(Dtemp,vs)

        diagEnergy= e
        diagEnergyCom= e
        sum1=0
        for i in range(len(diagEnergy)):
            sum1=sum1+diagEnergy[i]
            diagEnergyCom[i]=sum1

        copyEnergy=diagEnergyCom
        Information_preserved= 100*((diagEnergyCom[FinalFeatures-1])/diagEnergyCom[len(diagEnergyCom)-1])
        #print("information contained= ", 100*((diagEnergyCom[1])/diagEnergyCom[len(diagEnergyCom)-1]),"%")
        #print("PC1: ",vs[:,0],"\n")
        transformation_mat= np.array(vs[:,0])
        for i in range(FinalFeatures-1):
            #print("PC",i+2,": ",vs[:,i+1],"\n")
            transformation_mat=np.append(transformation_mat,vs[:,i+1],axis=0)

        transformation_mat=np.array(transformation_mat)
        reducedNF=FinalFeatures
        transformation_mat=np.reshape(transformation_mat,(reducedNF,len(vs[:,0])))

        return Information_preserved,transformation_mat
